Flutter国际化使用记录

首先在 pubspec.yaml 文件中加入

dev_dependencies:
  flutter_localizations:
    sdk: flutter
1
2
3

接着使用插件 vscode-flutter-i18n-json 官方教程

  1. Vscode 搜索 vscode-flutter-i18n-json 插件进行安装

  2. 使用 Vscode 快捷键 Ctrl+Shift+P 运行 Flutter I18n Json: Initialize 初始化

  3. 使用 Vscode 快捷键 Ctrl+Shift+P 运行 Flutter I18n Json: Add locale 创建语言

  4. 使用 Vscode 快捷键 Ctrl+Shift+P 运行 Flutter I18n Json: Update 更新语言

  5. 使用 Vscode 快捷键 Ctrl+Shift+P 运行 Flutter I18n Json: Remove locale 删除语言 (没用过，没图)。

初始化语言后，会在根目录生成一个 i18n的文件夹，里面会有一个 Json文件(默认是应该是英文,这里已经创建过一次中文简体的了。)

里面就是正常的Json格式

访问 I18n.of(context).home

手动切换语言 I18n.onLocaleChanged(new Locale('zh', 'CN'))

备注：每次修改Json文件都要使用 Vscode 快捷键 Ctrl+Shift+P 运行 Flutter I18n Json: Update 更新语言

最后附上 main.dart 文件代码 --- 官方地址

```
import 'package:flutter_localizations/flutter_localizations.dart';

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  final i18n = I18n.delegate;

  @override
  void initState() {
    super.initState();
    I18n.onLocaleChanged = onLocaleChange;
  }

  void onLocaleChange(Locale locale) {
    setState(() {
      I18n.locale = locale;
    });
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: app,
      localizationsDelegates: [
        i18n,
        GlobalMaterialLocalizations.delegate,
        GlobalWidgetsLocalizations.delegate,
        GlobalCupertinoLocalizations.delegate // <-- needed for iOS
      ],
      supportedLocales: i18n.supportedLocales,
     home: <your home widget>
    );
  }
}
```
